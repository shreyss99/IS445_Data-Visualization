# We see there are some values which are negative and we need to remove them

# for selecting rows that the value in 'Year Acquired' greater than 0

filter_2 = data['Year Constructed'] > 0
# for selecting rows that the value in 'Year Constructed' greater than 0

filter_3 = data['Age at Acquisition'] >= 0
# for selecting rows that the value in 'Age at Acquisition' greater than or equal to 0


data_filtered = data.loc[(filter_1)&(filter_2)&(filter_3)].copy()
# .loc[] select rows based on the filter (or filters) applied to the column (or columns)
